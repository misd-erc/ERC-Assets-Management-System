ERC-Assets-Management-System/WEB/
├── public/                          # Static assets served directly (HTML, icons, logos, manifest)
│   └── images/                      # Stores app logos, icons, and other image files
└── src/                             # Main source code folder (everything compiled by React + TypeScript)
    ├── api/                         # Handles API requests (Axios wrappers per module)
    ├── components/                  # Reusable UI components and feature-specific modules
    │   ├── auth/                    # Login, MFA, and authentication-related components
    │   ├── dashboard/               # Dashboard widgets (charts, quick actions, summaries)
    │   ├── layout/                  # Main app layout: sidebar, topbar, notifications, etc.
    │   ├── profile/                 # User profile, account settings, and privacy pages
    │   └── ui/                      # Common UI components (buttons, modals, forms, shadcn/ui)
    ├── hooks/                       # Custom React hooks for reusable logic
    │   ├── auth/                    # Hooks for login, logout, session handling
    │   └── data/                    # Hooks for fetching/managing data from APIs
    ├── lib/                         # Library configurations (e.g. Axios instance, interceptors)
    ├── pages/                       # Main pages/screens rendered by the router
    ├── services/                    # Business logic layer; combines API calls + rules
    ├── store/                       # Global state management (Zustand/Redux)
    │   ├── auth/                    # Auth state (tokens, user data)
    │   └── data/                    # App data state (assets, departments, etc.)
    ├── types/                       # TypeScript type definitions and interfaces
    │   ├── asset/                   # Types for asset management
    │   ├── audit/                   # Types for audit records
    │   ├── auth/                    # Types for authentication models
    │   ├── contract/                # Types for contract-related data
    │   ├── data/                    # Shared data types
    │   ├── ris/                     # Types for Request and Issuance Slip (RIS)
    │   ├── supply/                  # Types for supply management and stock cards
    │   └── user/                    # Types for user profiles and permissions
    └── utils/                       # Helper functions (dates, encryption, sanitization, etc.)
